  <script>
    const crtLoveTl = () => {
      const move = 1000;
      const heart = document.body;

      return new mojs.Timeline().add([
        new mojs.Tween({
          duration: move,
          onUpdate: (p) => {
            heart.style.transform = scale(${1 + 0.5 * p});
          },
          onComplete: () => {
            heart.style.opacity = 0;
            console.log("Boom!");
          }
        })
      ]);
    };

    const tl = crtLoveTl();
    tl.play();
  </script>
